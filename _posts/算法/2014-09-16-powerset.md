### 输出一个序列的powerset(所有子集)
1.用bit

```
#include <iostream>
using namespace std;

int main(){  
	const int size = 3;
	int S[size] = {1, 2, 3};    
	int total_sets = 1 << size;	//2^n

	for (int idx = 0; idx < total_sets; idx++){  
		int temp_idx = idx;  
		int element_id = 0;  
		cout<<"{";  
		while (temp_idx){  
			if (temp_idx & 1)	cout<<S[element_id];  
			temp_idx >>= 1;  
			++element_id;  
		}  
		cout<<"}"<<endl;  
	}  

	system("pause");
	return 0;  
}  
```

2.递归
一个递归法

```
#include <iostream>
using namespace std;
void build(char *str,int *tag,int n)
{
	if(n==5)
	{
		cout<<"{";
		for(int i=0;i<5;i++)
			if(tag[i]==1)
				cout<<str[i];
		cout<<"}"<<endl;
		return;
	}
	tag[n] = 0;
	build(str,tag,n+1);
	tag[n] = 1;
	build(str,tag,n+1);
}
int main()
{
	char a[5]={'a','b','c','d','e'};
	int tag[5];
	build(a,tag,0);
	return 0;
}
```
